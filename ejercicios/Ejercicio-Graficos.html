n <- 1:20

par(bg = "#fafafa", col="#555555")
plot(n, fib, pch = 21, cex = 1.5, lwd = 2, col = "#ADBF71", bg = "#4E63D0", 
     type = "b", lty = 4, xlab = "n", ylab = expression(F[n]), main = "Sucesion de Fibonacci")
